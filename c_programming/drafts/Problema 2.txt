#include <msp430.h>
#include <clock.h>

/**
 * main.c
 */




void trigger() {
    P1OUT |= BIT2; 

    TA1CCR0= 50;
    TA1CTL = TASSEL_2 |MC_1;
    while((TA1CCTL0 & CCIFG)==0);
    TA1CCTL0 &= ~CCIFG;
    P1OUT &=~BIT2;
    TA1CTL &= ~MC_1;

}

int main(void)


{
    WDTCTL = WDTPW | WDTHOLD;   // stop watchdog timer

    float dist=0;
    float tempo=0;
    int  dif=0, valora=0, valorb=0;


    P4DIR |=BIT7;
    P1DIR |=BIT0;

    P4OUT &=~BIT7;
    P1OUT &=~BIT0;

    P1DIR |=BIT2;
    P1OUT |=BIT2;

    P1DIR &= ~BIT3;
    P1SEL |=  BIT3;

    while(1){

    TA0CTL = TASSEL_2|MC_2;
    TA0CCTL2= CCIS_0|CM_1|CAP|SCS;
    TA0CTL |= TACLR;

    trigger();

    while((TA0CCTL2 & CCIFG)==0);
    valora=TA0CCR2;

    TA0CCTL2= CCIS_0|CM_2|CAP|SCS;
    while((TA0CCTL2 & CCIFG)==0);
    valorb=TA0CCR2;

    dif=valorb-valora;
    tempo= dif/1048576.;
    dist= (34000.*tempo)/2.;

    if (dist <20) {

        P1OUT |= BIT0;
        P4OUT &= ~BIT7;
        // Ligando LED VERMELHO
    }
    else if ((20 <= dist) && (dist <=40)){
        P4OUT |=  BIT7;
        P1OUT &= ~BIT0;
    }
    else {

        P4OUT |=  BIT7;
        P1OUT |=  BIT0;

    }

    TA2CCR0= 20000-1;
    TA2CTL = TASSEL_2 |MC_1;
    while((TA2CCTL0&CCIFG)==0);
    TA2CCTL0 &= ~CCIFG;



}
}
